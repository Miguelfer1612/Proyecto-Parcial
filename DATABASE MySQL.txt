CREATE DATABASE matricula;

USE matricula;

CREATE TABLE Profesor (
    id INT PRIMARY KEY,
    nombre VARCHAR(100),
    correo VARCHAR(100)
);

CREATE TABLE Curso (
    id VARCHAR(10) PRIMARY KEY,
    nombre VARCHAR(100),
    creditos INT
);
